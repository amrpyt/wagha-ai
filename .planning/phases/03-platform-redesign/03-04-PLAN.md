**Generation flow (bulk):**

```tsx
const handleGenerate = async () => {
  if (!selectedFile || selectedStyles.length === 0) return

  setCanvasState('generating')

  // Initialize results for each style
  const initialResults: SuggestionResult[] = selectedStyles.map(id => {
    const style = SUGGESTION_STYLES.find(s => s.id === id)!
    return {
      styleId: id,
      styleLabel: style.label,
      inputUrl: previewUrlRef.current || '',
      renderUrl: null,
      status: 'pending' as const,
    }
  })
  setResults(initialResults)
  setKeptResults(new Set())

  // Create project (first style creates project, subsequent use same project)
  // For simplicity, create one project with the first style
  const firstStyle = SUGGESTION_STYLES.find(s => s.id === selectedStyles[0])!

  try {
    // Create project
    const createFormData = new FormData()
    createFormData.append('file', selectedFile)
    createFormData.append('projectName', 'اقتراحات')
    createFormData.append('renderType', 'exterior')
    createFormData.append('template', firstStyle.template)
    createFormData.append('modifiers', JSON.stringify({}))
    createFormData.append('customPrompt', '')

    const createRes = await fetch('/api/project/create', {
      method: 'POST',
      body: createFormData,
    })

    if (!createRes.ok) throw new Error('فشل إنشاء المشروع')
    const { projectId } = await createRes.json()

    // Generate for each style in parallel
    const generationPromises = selectedStyles.map(async (styleId, index) => {
      const style = SUGGESTION_STYLES.find(s => s.id === styleId)!

      // Update status to generating
      setResults(prev => prev.map((r, i) => 
        i === index ? { ...r, status: 'generating' as const } : r
      ))

      try {
        // Call generate for this style
        // For simplicity, use SSE for each — in real impl, this would be batch API
        const result = await generateForStyle(projectId, style, selectedFile)
        
        setResults(prev => prev.map((r, i) => 
          i === index ? { ...r, renderUrl: result.renderUrl, status: 'complete' as const } : r
        ))
      } catch (err) {
        setResults(prev => prev.map((r, i) => 
          i === index ? { ...r, error: String(err), status: 'error' as const } : r
        ))
      }
    })

    await Promise.all(generationPromises)
    setCanvasState('complete')
  } catch (err) {
    setError(err instanceof Error ? err.message : 'خطأ غير معروف')
    setCanvasState('image-loaded')
  }
}

// Helper to generate for a single style
async function generateForStyle(projectId: string, style: StyleOption, file: File): Promise<{ renderUrl: string }> {
  return new Promise((resolve, reject) => {
    const eventSource = new EventSource(`/api/generate?projectId=${projectId}&template=${style.template}`)

    eventSource.onmessage = (e) => {
      const data = JSON.parse(e.data)
      if (data.error) {
        eventSource.close()
        reject(new Error(data.error))
        return
      }
      if (data.renderUrl) {
        eventSource.close()
        resolve({ renderUrl: data.renderUrl })
      }
    }

    eventSource.onerror = () => {
      eventSource.close()
      reject(new Error('انقطع الاتصال بالخادم'))
    }
  })
}
```

**Toggle kept result:**
```tsx
const toggleKept = (styleId: string) => {
  setKeptResults(prev => {
    const next = new Set(prev)
    if (next.has(styleId)) {
      next.delete(styleId)
    } else {
      next.add(styleId)
    }
    return next
  })
}
```
