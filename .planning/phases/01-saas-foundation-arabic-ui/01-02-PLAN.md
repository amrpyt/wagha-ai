## Objective

Implement complete authentication flows for Wagha-ai: email/password signup with email verification, login with session persistence, password reset via email link, logout from any page, and the email verification callback handler. All pages in Arabic RTL with Noor-UI components.

**Why this matters:** Auth is the gate to the entire application. Every feature requires a logged-in user.

**Output:** Working signup, login, email verification, password reset, and logout flows with Arabic RTL UI.

Create the auth Server Actions file with all authentication operations. This file uses `'use server'` directive and imports `createClient` from `@/lib/supabase/server`.

**signUp flow:**
1. Extract email and password from FormData
2. Validate: email is valid format, password is at least 8 chars
3. Create Supabase auth user with email/password
4. If successful, create a firm record (the user becomes the firm owner)
5. Add the user as an admin firm_member
6. Return success (Supabase will send verification email automatically)

**signIn flow:**
1. Extract email and password from FormData
2. Call `supabase.auth.signInWithPassword({ email, password })`
3. If error, return error message
4. If success, return success (page will revalidate and redirect)

**signOut flow:**
1. Call `supabase.auth.signOut()`
2. Revalidate the current path
3. Return

**resetPassword flow:**
1. Extract email from FormData
2. Call `supabase.auth.resetPasswordForEmail(email, { redirectTo: '/reset-password/confirm' })`
3. Return success

**verifyEmail flow:**
1. Extract token from searchParams
2. Call `supabase.auth.verifyOtp({ type: 'email', token })`
3. If successful, mark email as verified
4. Return success

```typescript
'use server'

import { createClient } from '@/lib/supabase/server'
import { redirect } from 'next/navigation'
import { revalidatePath } from 'next/cache'

// ================================================
// SIGN UP - creates user + firm + firm_member
// ================================================
export async function signUp(formData: FormData) {
  const supabase = await createClient()

  const email = formData.get('email') as string
  const password = formData.get('password') as string
  const firmName = formData.get('firmName') as string

  // Validate inputs
  if (!email || !password || !firmName) {
    return { error: 'جميع الحقول مطلوبة' }
  }

  if (password.length < 8) {
    return { error: 'كلمة المرور يجب أن تكون 8 أحرف على الأقل' }
  }

  if (!email.includes('@')) {
    return { error: 'البريد الإلكتروني غير صالح' }
  }

  // Create auth user
  const { data: authData, error: authError } = await supabase.auth.signUp({
    email,
    password,
    options: {
      data: {
        firm_name: firmName,
      },
    },
  })

  if (authError) {
    return { error: authError.message }
  }

  if (!authData.user) {
    return { error: 'فشل إنشاء الحساب' }
  }

  // Create firm record (user is the owner)
  const { error: firmError } = await supabase
    .from('firms')
    .insert({
      name: firmName,
      owner_id: authData.user.id,
      brand_color: '#1E3A5F',
    })

  if (firmError) {
    // Rollback: delete auth user
    await supabase.auth.admin.deleteUser(authData.user.id)
    return { error: 'فشل إنشاء الشركة' }
  }

  // Get the firm we just created
  const { data: firm } = await supabase
    .from('firms')
    .select('id')
    .eq('owner_id', authData.user.id)
    .single()

  if (firm) {
    // Add user as admin firm_member
    await supabase.from('firm_members').insert({
      firm_id: firm.id,
      user_id: authData.user.id,
      role: 'admin',
    })
  }

  // Supabase sends verification email automatically
  return { success: true, needsVerification: true }
}

// ================================================
// SIGN IN
// ================================================
export async function signIn(formData: FormData) {
  const supabase = await createClient()

  const email = formData.get('email') as string
  const password = formData.get('password') as string

  if (!email || !password) {
    return { error: 'البريد الإلكتروني وكلمة المرور مطلوبان' }
  }

  const { error } = await supabase.auth.signInWithPassword({
    email,
    password,
  })

  if (error) {
    return { error: 'البريد الإلكتروني أو كلمة المرور غير صحيحة' }
  }

  revalidatePath('/dashboard')
  redirect('/dashboard')
}

// ================================================
// SIGN OUT
// ================================================
export async function signOut() {
  const supabase = await createClient()
  await supabase.auth.signOut()
  revalidatePath('/')
  redirect('/login')
}

// ================================================
// RESET PASSWORD
// ================================================
export async function resetPassword(formData: FormData) {
  const supabase = await createClient()

  const email = formData.get('email') as string

  if (!email) {
    return { error: 'البريد الإلكتروني مطلوب' }
  }

  const { error } = await supabase.auth.resetPasswordForEmail(email, {
    redirectTo: `${process.env.NEXT_PUBLIC_SITE_URL}/reset-password/confirm`,
  })

  if (error) {
    return { error: error.message }
  }

  return { success: true }
}

// ================================================
// VERIFY EMAIL (for confirmation callback)
// ================================================
export async function verifyEmail(token: string) {
  const supabase = await createClient()

  const { error } = await supabase.auth.verifyOtp({
    type: 'email',
    token,
  })

  if (error) {
    return { error: 'فشل التحقق من البريد الإلكتروني' }
  }

  return { success: true }
}
```

**src/components/auth/AuthForm.tsx** - Shared form wrapper with state handling:
```typescript
'use client'

import { useActionState } from 'react'
import { Button, Input } from 'noorui-rtl'
import type { ReactNode } from 'react'

interface FormField {
  name: string
  label: string
  type?: string
  placeholder?: string
  required?: boolean
}

interface AuthFormProps {
  action: (formData: FormData) => Promise<any>
  fields: FormField[]
  submitLabel: string
  children?: ReactNode
  initialState?: any
}

export function AuthForm({ action, fields, submitLabel, children, initialState }: AuthFormProps) {
  const [state, formAction, isPending] = useActionState(action, initialState)

  return (
    <form action={formAction} className="space-y-4 rtl:mt-6">
      {fields.map((field) => (
        <div key={field.name} className="rtl:space-y-2">
          <label
            htmlFor={field.name}
            className="block text-sm font-medium text-right rtl:text-right"
          >
            {field.label}
            {field.required && <span className="rtl:mr-1 text-red-500">*</span>}
          </label>
          <Input
            id={field.name}
            name={field.name}
            type={field.type || 'text'}
            placeholder={field.placeholder}
            required={field.required}
            disabled={isPending}
            className="w-full rtl:text-right"
          />
        </div>
      ))}

      {state?.error && (
        <div className="bg-red-50 rtl:border-r-4 rtl:border-red-500 rtl:p-3 rtl:rounded">
          <p className="text-sm text-red-700 text-right">{state.error}</p>
        </div>
      )}

      {state?.success && (
        <div className="bg-green-50 rtl:border-r-4 rtl:border-green-500 rtl:p-3 rtl:rounded">
          <p className="text-sm text-green-700 text-right">
            {state.needsVerification
              ? 'تم إرسال رابط التحقق إلى بريدك الإلكتروني'
              : state.success}
          </p>
        </div>
      )}

      <Button type="submit" disabled={isPending} className="w-full rtl:mt-4">
        {isPending ? 'جاري التحميل...' : submitLabel}
      </Button>

      {children}
    </form>
  )
}
```

**src/app/api/auth/callback/route.ts** - Email verification callback (handles Supabase email confirm links):
```typescript
import { NextRequest, NextResponse } from 'next/server'
import { createClient } from '@/lib/supabase/server'

export async function GET(request: NextRequest) {
  const requestUrl = new URL(request.url)
  const token = requestUrl.searchParams.get('token')
  const type = requestUrl.searchParams.get('type')
  const next = requestUrl.searchParams.get('next') || '/dashboard'

  if (type === 'email_confirm' && token) {
    const supabase = await createClient()

    const { error } = await supabase.auth.verifyOtp({
      type: 'email',
      token,
    })

    if (!error) {
      return NextResponse.redirect(new URL(next, requestUrl.origin))
    }
  }

  // Redirect to login on error
  return NextResponse.redirect(new URL('/login?error=verification_failed', requestUrl.origin))
}
```
