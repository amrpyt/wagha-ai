```typescript
import { createClient } from '@/lib/supabase/server'
import { revalidatePath } from 'next/cache'
import type { FirmMember, Invitation } from '@/types/database.types'

// ================================================
// ACCOUNT SETTINGS
// ================================================
export async function updateAccount(data: { name?: string; email?: string }): Promise<{ success: boolean; error?: string }> {
  const supabase = await createClient()

  const { data: { user } } = await supabase.auth.getUser()
  if (!user) {
    return { success: false, error: 'غير مصرح' }
  }

  const updates: { data?: { full_name?: string }; email?: string } = {}

  if (data.name) {
    updates.data = { full_name: data.name }
  }

  if (data.email && data.email !== user.email) {
    // Email change requires re-verification
    const { error } = await supabase.auth.updateUser({
      email: data.email,
    })

    if (error) {
      return { success: false, error: error.message }
    }

    return { success: true } // Will need email verification
  }

  if (Object.keys(updates).length > 0) {
    const { error } = await supabase.auth.updateUser(updates)

    if (error) {
      return { success: false, error: error.message }
    }
  }

  revalidatePath('/settings')
  return { success: true }
}

export async function changePassword(
  currentPassword: string,
  newPassword: string
): Promise<{ success: boolean; error?: string }> {
  const supabase = await createClient()

  if (newPassword.length < 8) {
    return { success: false, error: 'كلمة المرور يجب أن تكون 8 أحرف على الأقل' }
  }

  // Verify current password by trying to sign in
  const { data: { user } } = await supabase.auth.getUser()
  if (!user?.email) {
    return { success: false, error: 'غير مصرح' }
  }

  const { error: signInError } = await supabase.auth.signInWithPassword({
    email: user.email,
    password: currentPassword,
  })

  if (signInError) {
    return { success: false, error: 'كلمة المرور الحالية غير صحيحة' }
  }

  const { error } = await supabase.auth.updateUser({
    password: newPassword,
  })

  if (error) {
    return { success: false, error: error.message }
  }

  return { success: true }
}

export async function updateFirm(
  data: { name?: string; logoUrl?: string; brandColor?: string }
): Promise<{ success: boolean; error?: string }> {
  const supabase = await createClient()

  const { data: { user } } = await supabase.auth.getUser()
  if (!user) {
    return { success: false, error: 'غير مصرح' }
  }

  // Get user's firm
  const { data: firmMember } = await supabase
    .from('firm_members')
    .select('firm_id, role')
    .eq('user_id', user.id)
    .single()

  if (!firmMember) {
    return { success: false, error: 'ليست عضو في شركة' }
  }

  // Only admins can update firm settings
  if (firmMember.role !== 'admin') {
    return { success: false, error: 'فقط المشرف يمكنه تحديث إعدادات الشركة' }
  }

  const updates: Record<string, string> = {}
  if (data.name) updates.name = data.name
  if (data.logoUrl !== undefined) updates.logo_url = data.logoUrl
  if (data.brandColor) updates.brand_color = data.brandColor

  const { error } = await supabase
    .from('firms')
    .update(updates)
    .eq('id', firmMember.firm_id)

  if (error) {
    return { success: false, error: 'فشل تحديث إعدادات الشركة' }
  }

  revalidatePath('/settings')
  revalidatePath('/settings/firm')
  revalidatePath('/dashboard')
  return { success: true }
}

export async function uploadLogo(file: File): Promise<{ success: boolean; url?: string; error?: string }> {
  const supabase = await createClient()

  const { data: { user } } = await supabase.auth.getUser()
  if (!user) {
    return { success: false, error: 'غير مصرح' }
  }

  // Get user's firm
  const { data: firmMember } = await supabase
    .from('firm_members')
    .select('firm_id')
    .eq('user_id', user.id)
    .single()

  if (!firmMember) {
    return { success: false, error: 'ليست عضو في شركة' }
  }

  // Validate file
  if (!file.type.startsWith('image/')) {
    return { success: false, error: 'يجب أن يكون الملف صورة' }
  }

  if (file.size > 2 * 1024 * 1024) { // 2MB limit
    return { success: false, error: 'حجم الملف يجب أن يكون أقل من 2 ميجابايت' }
  }

  // Generate unique filename
  const ext = file.name.split('.').pop()
  const filename = `${firmMember.firm_id}/logo-${Date.now()}.${ext}`

  // Upload to Supabase Storage
  const { error: uploadError } = await supabase.storage
    .from('firm-logos')
    .upload(filename, file, {
      cacheControl: '3600',
      upsert: true,
    })

  if (uploadError) {
    return { success: false, error: 'فشل رفع الشعار' }
  }

  // Get public URL
  const { data: { publicUrl } } = supabase.storage
    .from('firm-logos')
    .getPublicUrl(filename)

  return { success: true, url: publicUrl }
}

export async function inviteTeamMember(
  email: string,
  role: 'admin' | 'member' = 'member'
): Promise<{ success: boolean; error?: string }> {
  const supabase = await createClient()

  const { data: { user } } = await supabase.auth.getUser()
  if (!user) {
    return { success: false, error: 'غير مصرح' }
  }

  // Get user's firm and check if admin
  const { data: firmMember } = await supabase
    .from('firm_members')
    .select('firm_id, role')
    .eq('user_id', user.id)
    .single()

  if (!firmMember || firmMember.role !== 'admin') {
    return { success: false, error: 'فقط المشرف يمكنه دعوة أعضاء الفريق' }
  }

  // Check if already a member
  const { data: existingMember } = await supabase
    .from('firm_members')
    .select('id')
    .eq('firm_id', firmMember.firm_id)
    .eq('user_id', (await supabase.auth.getUser()).data.user?.id)
    .single()

  // Check if user already exists in this firm
  const { data: existingEmailUser } = await supabase
    .from('firm_members')
    .select('id, user_id')
    .eq('firm_id', firmMember.firm_id)
    .maybeSingle()

  if (existingEmailUser) {
    return { success: false, error: 'هذا البريد الإلكتروني موجود بالفعل في الفريق' }
  }

  // Generate invite token
  const token = crypto.randomUUID()

  // Create invitation record
  const { error: inviteError } = await supabase
    .from('invitations')
    .insert({
      firm_id: firmMember.firm_id,
      email,
      role,
      invited_by: user.id,
      token,
    })

  if (inviteError) {
    return { success: false, error: 'فشل إرسال الدعوة' }
  }

  // Send invite email via Supabase Auth
  const adminClient = createAdminClient()
  const { error: authError } = await adminClient.auth.admin.inviteUserByEmail(email, {
    data: {
      firm_id: firmMember.firm_id,
      role,
    },
    redirectTo: `${process.env.NEXT_PUBLIC_SITE_URL}/invite/accept?token=${token}`,
  })

  if (authError) {
    // Clean up invitation if email failed
    await supabase.from('invitations').delete().eq('token', token)
    return { success: false, error: 'فشل إرسال البريد الإلكتروني للدعوة' }
  }

  revalidatePath('/settings')
  return { success: true }
}

export async function getTeamMembers(): Promise<FirmMember[]> {
  const supabase = await createClient()

  const { data: { user } } = await supabase.auth.getUser()
  if (!user) return []

  const { data: firmMember } = await supabase
    .from('firm_members')
    .select('firm_id')
    .eq('user_id', user.id)
    .single()

  if (!firmMember) return []

  const { data: members } = await supabase
    .from('firm_members')
    .select('*, auth.users(email)')
    .eq('firm_id', firmMember.firm_id)

  return members || []
}

export async function getInvitations(): Promise<Invitation[]> {
  const supabase = await createClient()

  const { data: { user } } = await supabase.auth.getUser()
  if (!user) return []

  const { data: firmMember } = await supabase
    .from('firm_members')
    .select('firm_id')
    .eq('user_id', user.id)
    .single()

  if (!firmMember) return []

  const { data: invitations } = await supabase
    .from('invitations')
    .select('*')
    .eq('firm_id', firmMember.firm_id)
    .eq('status', 'pending')

  return invitations || []
}
```

```typescript
import { useActionState, useRef } from 'react'

import { updateFirm, uploadLogo } from '@/lib/actions/settings'

const initialState = {}

export function FirmSettingsForm({ firm }: { firm: { name: string; logo_url: string | null; brand_color: string } }) {
  const [state, formAction, isPending] = useActionState(updateFirm, initialState)
  const fileInputRef = useRef<HTMLInputElement>(null)
  const [logoUrl, setLogoUrl] = useState(firm.logo_url)
  const [isUploading, setIsUploading] = useState(false)

  async function handleLogoUpload(e: React.ChangeEvent<HTMLInputElement>) {
    const file = e.target.files?.[0]
    if (!file) return

    setIsUploading(true)
    const result = await uploadLogo(file)

    if (result.success && result.url) {
      setLogoUrl(result.url)
      // Update firm with new logo URL
      const formData = new FormData()
      formData.append('logoUrl', result.url)
      formAction(formData)
    }

    setIsUploading(false)
  }

  return (
    <form action={formAction} className="rtl:space-y-6">
      {/* Logo */}
      <div>
        <label className="block text-sm font-medium text-gray-700 rtl:text-right rtl:mb-2">
```

```typescript
  const supabase = await createClient()
```

```typescript
import { createClient } from '@/lib/supabase/server'
import { redirect } from 'next/navigation'
import { getTeamMembers, getInvitations } from '@/lib/actions/settings'
import { TeamInviteForm } from '@/components/settings/TeamInviteForm'
import { TeamMemberList } from '@/components/settings/TeamMemberList'

export default async function TeamSettingsPage() {
  const supabase = await createClient()
  const { data: { user } } = await supabase.auth.getUser()

  if (!user) {
    redirect('/login')
  }

  // Check if user is admin
  const { data: firmMember } = await supabase
    .from('firm_members')
    .select('role')
    .eq('user_id', user.id)
    .single()

  if (!firmMember || firmMember.role !== 'admin') {
    redirect('/settings')
  }

  const members = await getTeamMembers()
  const invitations = await getInvitations()

  return (
    <div className="max-w-2xl">
      {/* Breadcrumb */}
      <div className="rtl:mb-6">
        <h1 className="text-2xl font-semibold text-gray-900 rtl:text-right">
          إدارة الفريق
        </h1>
        <p className="text-gray-500 rtl:text-right rtl:mt-1">
          دعوة أعضاء جدد للانضمام إلى شركتك
        </p>
      </div>

      {/* Invite Form */}
      <div className="bg-white rtl:rounded-lg rtl:border rtl:border-gray-200 rtl:p-6 rtl:mb-6">
        <h2 className="text-lg font-semibold rtl:text-right rtl:mb-4">
          دعوة عضو جديد
        </h2>
        <TeamInviteForm />
      </div>

      {/* Pending Invitations */}
      {invitations.length > 0 && (
        <div className="bg-yellow-50 rtl:rounded-lg rtl:border rtl:border-yellow-200 rtl:p-4 rtl:mb-6">
          <h3 className="font-medium text-yellow-800 rtl:text-right rtl:mb-2">
            دعوات معلقة
          </h3>
          <ul className="rtl:space-y-2">
            {invitations.map((inv) => (
              <li key={inv.id} className="rtl:flex rtl:justify-between rtl:text-sm text-yellow-700">
                <span>{inv.email}</span>
                <span>{inv.role === 'admin' ? 'مشرف' : 'عضو'}</span>
              </li>
            ))}
          </ul>
        </div>
      )}

      {/* Current Members */}
      <div className="bg-white rtl:rounded-lg rtl:border rtl:border-gray-200 rtl:p-6">
        <h2 className="text-lg font-semibold rtl:text-right rtl:mb-4">
          أعضاء الفريق الحاليون
        </h2>
        <TeamMemberList members={members} />
      </div>
    </div>
  )
}
```
